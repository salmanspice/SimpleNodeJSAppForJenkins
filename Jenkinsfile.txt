pipeline {
  // agent any
  agent {label 'Jenkins-slave'}  
  tools {nodejs "node"}
    
  stages {
        
    stage('Git') {
      steps {
        script {
           // The below will clone your repo and will be checked out to master branch by default.
           git credentialsId: 'github-jenkins', url: 'https://github.com/salmanspice/SimpleNodeJSAppForJenkins'
           // Do a ls -lart to view all the files are cloned. It will be clonned. This is just for you to be sure about it.
           sh "ls -lart ./*" 
           // List all branches in your repo. 
           sh "git branch -a"
          }
      }
    }
     
    stage('Build') {
      steps {
        sh 'pwd'
        sh 'npm install'
        sh 'npm install express'
      }
    }      
     stage('Deployment') {
        steps {
            script {
                sshPublisher(publishers: [
                    sshPublisherDesc(
                        configName: 'live-prod',
                        transfers: [
                            sshTransfer(
                                sourceFiles: "index.js",
                                execCommand: "cd /var/lib/jenkins/workspace/nodeapp; npm install; npm install express; export BUILD_ID=dontKillMe; nohup node index.js & ",
                                remoteDirectory: '/var/www/www.printerpix.com' 
                            )
                        ],
                        usePromotionTimestamp: false,
                        useWorkspaceInPromotion: false,
                        verbose: false
                    )]
                )
            }
        }
    }       
  }
}
